import zipfile
import os

# 此腳本用於自動將原始圖片重新命名並打包，以供 GitHub Pages 使用
input_files = [

    'input_file_3.png', 'input_file_4.png', 'input_file_5.png'
]

zip_filename = 'the_beast_images.zip'

with zipfile.ZipFile(zip_filename, 'w') as zipf:
    for i in range(len(input_files)):
        src = input_files[i]
        dst = output_names[i]
        if os.path.exists(src):
            zipf.write(src, arcname=dst)
